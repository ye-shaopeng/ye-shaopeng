# `函数`

```javascript
const mapping = {
    // software
    '04': {
        '001': 'obsidian',
        '002': 'vscode',
        '003': 'idea'
    }
};

const match = title.match(/^\[m\](.+)  ([0-9]{2})-([0-9]{3})(.+)$/);
if (match) {
    const heading = match[1];
    const subject = match[2];
    const course = match[3];
    const rest = match[4];
    if (mapping[subject] && mapping[subject][course]) {
        return `${heading}\n${subject}-${mapping[subject][course]}${rest}`;
    }
}
if (title.match(/^\[m\].+$/)) {
    title = title.replace("[m]", "");
}
return title.replace("  ", "\n");
```
